// Author:PanDaoxi
#include <iostream>
using namespace std;
int dp[101][101];
void lcs(string & s1,string & s2){
    int m=s1.size();
    int n=s2.size();
    s1="#"+s1;
    s2="#"+s2;
    for(int i=1;i<=m;i++){
        for(int j=1;j<=n;j++){
            if(s1[i]==s2[j]) dp[i][j]=dp[i-1][j-1]+1; 
            else dp[i][j]=max(dp[i-1][j],dp[i][j-1]);     
        }
    } 
    cout<<dp[m][n]<<endl;
} 
int main(){
    string s1,s2;
    cin>>s1>>s2;
    lcs(s1,s2); 
    return 0;
}
